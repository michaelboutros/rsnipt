  def initialize(command, arguments)
    self.extend(SniptCLICommands)
        
    if arguments.last == '--help'
      send(command.to_sym, arguments)
      exit
    elsif arguments.length == 0 && !COMMANDS.include?(command)
      send('exec', [command])
      exit
    else
      send(command.to_sym, arguments)
      exit
    end
  end
