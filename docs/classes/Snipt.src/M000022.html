    def snipts(tag = nil, client = nil)
      tag_path = "/tag/#{tag}" if tag
      
      begin
        snipts_page = WWW::Mechanize.new.get("http://snipt.net/public/#{tag_path}")      
        snipts = snipts_page.search('ul.snipts/li').collect {|snipt| parse_snipt(snipt, client)}
      rescue
        []
      end
    end
